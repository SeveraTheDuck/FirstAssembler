:GetCirclePosition
        in
        pop rax
        in
        pop rbx

:GetCircleRadius
        in
        pop rcx

        ret

; Check wether the point is in the circle
:CheckPoint
        push rdx
        push 32
        div_module
        push rax
        sub
        square

        push rdx
        push 32
        div
        push rbx
        sub
        square

        add
        push rcx
        square

        jae :MakePoint

:NextIteration
        push rdx
        push 1
        add
        pop rdx

        push rdx
        push 1024
        ja :CheckPoint

        ret

; Write in RAM if the point is in circle
:MakePoint
        push 1
        pop [rdx]
        jmp :NextIteration
